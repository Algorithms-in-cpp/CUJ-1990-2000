/*      TESTS MICROSOFT'S VERSION OF QUICKSORT     */
/* =============================================== */
#include <stdlib.h>
#include "tsrtdefs.h"
void
main(int argc, char **argv)
{
    SORT_TEST_STRU SortParms;

    SortParms.argc      = argc;
    SortParms.argv      = argv;
    SortParms.SortFun   = qsort;
    SortParms.OKCodes   = (char *) NULL;
    SortParms.ExecName  = "mqsort";
    SortParms.ExecLabel = "Microsoft Qsort";

    tstsort(&SortParms);
}
