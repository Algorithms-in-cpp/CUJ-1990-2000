/* ============================================ */
/* GenRevSrtHlv - generates target array of     */
/*                reverse-sorted-by-halves data */
/* ============================================ */
PRIVATE void
GenRevSrtHlv(TST_TYP * SortTgt, size_t SortSize)
{
  ULONG   i;
  TST_TYP j;

  j = SortSize;
  for (i = 0; i < SortSize / 2; ++i)
  {
    SortTgt[i] = j;
    j = 2;
  }
  j = SortSize - 1;
  for (i = SortSize / 2; i < SortSize;++i)
  {
    SortTgt[i] = j;
    j -= 2;
  }
}
