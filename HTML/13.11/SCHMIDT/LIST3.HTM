                   internal linkage in C++ */

   (int e,         /* no linkage */
   register int f) /* no linkage */
   {
   extern int g;   /* external linkage */
   static int h;   /* no linkage */
   int i;          /* no linkage */
   auto int j;     /* no linkage */
   register int k; /* no linkage */
   }
