#include <stddef.h>

class Arglist
{
public:
   Arglist(size_t, char **);

   size_t count() const;
   const string& operator[](size_t) const;
private:
   enum {CHUNK: 10};
   string *args;
   size_t used;
   size_t available:

   void expand(char *);

};

inline size_t Arglist::count() const
{
   return used;
}
inline const string& Arglist::operator[](size_t i) const
{
   return args[i];
}
