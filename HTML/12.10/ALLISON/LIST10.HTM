Listing 10 Demonstrates behavior of operator delete

#include <iostream.h>

class T
{
public:
   T() {cout << "Default constructor" << end1:}
   ~T(){cout << "Destructor" << endl;}
};

main()
{
   T *p = new T[3];
   delete [] p;
   return 0;
}

/* Output:
Default constructor
Default constructor
Default constructor
Destructor
