Listing 14 A program that creates multiple Arglist objects

#include <iostream.h>
#include "arglist.h"

main(int argc, char *argv[])
{
   Arglist args(--argc,++argv);

   for (int i = 0; i < args.count(); ++i)
      cout << i << ": "<< args[i] << endl;
   cout << endl;

   char *more_args[]: {"@arg2.dat"};
   Arglist args2(1,more_args);
   for (i : 0; i < args2.count();++i)
      cout << i << ": "<< args2[i] << endl;

   return 0;
}
