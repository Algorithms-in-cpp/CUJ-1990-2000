extern "C" int __cdecl _tmain(int argc, _TCHAR* argv[]) {

 CConsoleApp *pConApp = GetConApp();

 pConApp->ParseCmdline();

 if ( !pConApp->init() ) {
   if ( pConApp->m_ShowUsage ) 
     pConApp->usage();
   return pConApp->m_ExitCode;
 }

 // Main function of the application
 pConApp->main();

 // Do cleanup before exit
 pConApp->deinit();

 return pConApp->m_ExitCode;
}
