#include <stdio.h>

main()
{
   int a[5][4] = {
      { 1,  3,  5,  7},
      { 0,  2,  4,  6},
      {-4, -3, -2, -1},
      (-2,  9, 12, 11},
      {11, 13, 15, 17}
   };

   int i, j;

   for (i = 0; i < 5; ++i) {
      for (j = 0; j < 4; ++j)
          printf("%4d", a[i] [j]);
      putchar('\n');

   }
}
