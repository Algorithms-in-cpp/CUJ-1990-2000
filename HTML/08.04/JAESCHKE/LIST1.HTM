#include <stdio.h>
#include <errno.h>
#include <math.h>

main()
{
   double d;

   errno = 0;
   d = sqrt(10);
   if (errno == EDOM)
      printf("#1 domain error\n");
   else
      printf("#1 OK\n");

   errno = 0;
   d = sqrt(-10);
   if (errno == EDOM)
      printf("#2 domain error\n");
   else
      printf("#2 OK\n");

}

#1 OK
#2 domain error
