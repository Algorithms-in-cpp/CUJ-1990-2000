/* tclock.c:                         *
 * Echo clock ticks until user abort */
#include <stdio.h>
#include "clock.h"

void main()
{
   unsigned long last = 0L, now;

   clock_reset();
   for (;;)
   {
      now = clock_elapsed();
      if (last != now)
      {
         printf("%ld\n",now);
         last = now;
      }
   }
}
