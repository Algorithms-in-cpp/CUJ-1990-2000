Listing 4 Video constants and macros

/* video.h */

/* Basic Foreground colors *
 * (black background)      */
#define BLACK        0
#define BLUE         1
#define GREEN        2
#define CYAN         3
#define RED          4
#define MAGENTA      5
#define BROWN        6
#define LT_GRAY      7
#define DK_GRAY      8
#define LT_BLUE      9
#define LT_GREEN     10
#define LT_CYAN      11
#define LT_RED       12
#define LT_MAGENTA   13
#define YELLOW       14
#define WHITE        15

#define NORMAL LT_GRAY

/* The following are for *
 *monochrome systems     */
#define NO_DISPLAY 0
#define UNDERLINE 1
#define REVERSE 0x70

/* This makes a basic foreground *
 * color a background color      */
#define bkgr(c) ((unsigned) c << 4)

/* This turns on the high *
 * intensity bit          */
#define bright(c) (c | 0x08)

/* This turns on the blink bit */
#define blink(c) (c | 0x80)
