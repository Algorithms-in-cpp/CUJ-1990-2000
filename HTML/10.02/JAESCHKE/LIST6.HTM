#include <stdio.h>

main()
{
      int a = 8, b = 2, c = 6, d = 3;

      push(a);
      push(b);
      add();          /* t1 = a + b */
      push(c);
      push(d);
      subtract();     /* t2 = c - d */
      multiply();     /* t3 = t1 * t2 */

      printf("Result = %d\n", pop());

      return 0;
}

Result = 30
