void find_min_max(one, two, three, minimum, maximum)
   int one, two, three;  /* Input values to compare */
   int *minimum;
   int *maximum;
      {
      if (one > two)
         {
         if (one > three)
            *maximum = one;
         else
            *maximum = three;
         if (two > three)
            *minimum = three;
         else
            *minimum = two;
      else
         {
         if (two > three)
            *maximum = two;
         else
            *maximum = three;
         if (one > three)
            *minimum = three;
         else
            *minimum = one;
         }
      return ;
      }
