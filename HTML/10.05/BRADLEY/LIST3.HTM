#include <conio.h>

#include "labmastr.h"

void disable_adc()
/*& Disable adc through stopping timer 5's output. */
       {
       timer5_off();
       }

void enable_adc()
/*& Enable adc through starting timer 5's output. */
       {
       timer5_on();
       }

int adc_ status()
/*& Return the ADC status */
       {
       return (inp(ADC_STATUS));
       }
