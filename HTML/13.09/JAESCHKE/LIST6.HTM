Listing 6 Calling localeconv

#include <stdio.h>
#include <locale.h>
#include "locnames.h"

main()
{
      struct lconv *pst;

      setlocale(LC_ALL, LOC_French);
      pst = localeconv();

      printf("Locale: %s\n", STR(LOC_French));
      printf("decimal_point:   %s\n", pst->decimal_point);
      printf("int_curr_symbol: %s\n", pst->int_curr_symbol);

      return 0;
}
