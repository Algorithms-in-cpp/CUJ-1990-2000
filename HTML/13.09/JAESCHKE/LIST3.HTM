Listing 3 Printing Macro Names for Locales

#include <locale.h>
#include "locnames.h"
#include <stdio.h>

main()
{
      char *ploc_str;

      ploc_str = setlocale(LC_ALL, LOC_German);
      if (ploc_str == NULL) {
            printf("Can't establish locale %s\n", STR(LOC_German));
            return 0;
      }

      printf("Established locale %s\n", STR(LOC_German));

      return 0;
}
