Listing 3: Sorting an array of float


// Listing 3: test1.cpp
#include <iostream.h>
#include "qsortT.h"
int main()
{
  float vlArry[] =
  { 8056, 10078.2, 1006.79, 90023 };
  int n = sizeof(vlArry)/sizeof(float);

  qsortVT(vlArry, n);
  for (int i = 0; i < n ; i++)
    cout << vlArry[i] << endl;
  return 0;
}
