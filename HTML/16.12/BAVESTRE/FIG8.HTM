#include "stdafx.h"

CSimple3DApp Simple3DApp;

BOOL CSimple3DApp::InitInstance()
{
    m_pMainWnd=new CMainWindow;
    m_pMainWnd->ShowWindow(m_nCmdShow);
    m_pMainWnd->UpdateWindow();
    return TRUE;
}

CMainWindow::CMainWindow()
{
    CreateEx(0,"MyOpenGL","Simple3D",
        WS_OVERLAPPEDWINDOW,
        100,100,300,300,0,0);
}
