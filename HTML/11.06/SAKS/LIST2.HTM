Listing 2 A class template for a dynamic array

// da.h

#include <iostream.h>

template <class T> class array
   {
public:
   array(size_t n = 0};
   array(const array<T> &fa);
   ~array();
   array<T> &operator=
          (const array<T> &fa);
   T &operator[](size_t i) const;
   size_t length() const;
private:
   T *array;
   size_t len;
   };
