Listing 2 Prompts for a integer, echoes its value and address

// int.cpp: Prompt for an integer
# include <iostream.h>

main()
{
   int i;

   cout << ''Please enter an integer: '';
   cin >> i;
   cout << ''i == '' << i << '\n';
   cout << ''&i == '' << &i << '\n';
   return 0;
}

// Sample Execution:
// Please enter an integer: 10
// i == 10
// &i == Oxfff4
