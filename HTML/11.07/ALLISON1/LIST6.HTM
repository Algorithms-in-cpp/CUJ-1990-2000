//base1.cpp: Shows the bases  of integers
#include <iostreams.h>

main()
{
   int x, y, z;

   cout << "Enter three ints: ";
   cin >> x >> y >> z;
   cout << x << ',' << y << ',' << z << endl;

   // Print in different bases
   cout << x << ',';
   cout.setf(ios::oct,ios::basefield);
   cout << y << ',';
   cout.setf(ios::hex,ios::basefield);
   cout << z << endl;

   // Show the base prefix
   cout.setf(ios::showbase);
   cout << X << ',';
   cout.setf(ios::oct,ios::basefield);
   cout << y << ',';
   cout.setf(ios::hex,ios::basefield);
   cout << z << endl;
   return 0;
}
