// width.cpp:   Control width of input strings
#include <iostream.h>
#include <iomanip.h>

main()
{
   char s1[10], s2[3], s3[20];

   cin >> setw(10) >> s1
      >> setw(3) >> s2
      >> s3;

   cout << s1 << ',' << s2 << ',' << s3 << endl;
   return 0;
}
