Listing 4 mdadll.bat — builds the Windows DLL

rem Builds: MDAWIN.DLL
rem Dynamic link library for Windows.
rem
bcc -c -WD -D_RTLDLL -ml! mstream.cpp libmain.cpp
tlink x/c/C/Twd cOdl mstream libmain,mdawin.dll,,
    import crtldll,mdawin.def
implib mdawin.lib mdawtn.dll
bcc -c -WS -D_RTLDLL -ml moutinst.cpp
tlib mdawin.lib +moutinst.obj
