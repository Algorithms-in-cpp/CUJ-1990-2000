FUNCTION main
? sum (1, 2, 3, 4, 5)  // Should be 15

? sum()                // Should be zero

? sum ("ABC")          // Undefined

? reverse ("ABC")      // Should be "CBA"

? reverse ("")         // Should be zero-length string

? peek (64, 0)         // Beginning of BIOS data area

? lrc ("ABCDEFG")      // Should be 64 
RETURN NIL
