class shape
       {
public:
       enum palette { BLUE, GREEN, RED };
       shape(palette c);
       virtual double area() const;
       virtual const char *name() const;
       virtual ostream &put(ostream &os) const;
       virtual palette color() const;
private:
       palette _color;
       static const char *color_image[RED - BLUE + 1];
       };
