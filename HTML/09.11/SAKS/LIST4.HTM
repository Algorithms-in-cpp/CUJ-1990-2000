#include <stdio.h>

void put(char c, FILE *stream)
       {
       fputc(c, stream);
       }

void put(const char *s, FILE *stream)
       {
       fputs(s, stream);
       }

int main()
       {
       void put(char, FILE *);
       put("hello, world", stdout);  // error
       put('\n', stdout);            // ok
       return 0;
       }
