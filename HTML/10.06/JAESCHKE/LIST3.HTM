#include <stdio.h>

#define MAX_LEN 10
#define EDIT_MASK(size) "%" #size "s"

main()
{
       char str [MAX_LEN + 1];
       
       printf("Enter string (max length %u): ", MAX_LEN);
       scanf(EDIT_MASK(MAX_LEN), str);
       printf("str contains >%s<\n", str);
       
       return (0)
}
