#
# Make file for ShowTch demonstration program.
#
.c.obj:
   CL /c /W3 /Oit /D LINT_ARGS $*.c

GetTouch.obj: $*.c \
             dvrfunc.h \
             GetTouch.h

ShowTch.obj:  $*.c \
             dvrfunc.h \
             GetTouch.\

ShowTch.exe: $*.  \
      dvrfunc.obj \
      GetTouch.obj \
      $*.obj
   Link ShowTch+GetTouch/E/NOI, , NUL, DvrFunc;
       ShowTch
