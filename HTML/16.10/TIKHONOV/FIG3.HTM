struct SQLVAR    {
    short sqltype;
    unsigned short sqllen;
    char* sqldata;

struct SQLDA    {
    char sqldaid[8];
    long sqldabc;
    short sqln;
    short sqld;       // number of fields
    SQLVAR sqlvar[1];
