Listing 4: Demangling function names

#include "hungary.h"
#include "bscapi.h"
#include <stdio.h>
#include <stdlib.h>

const char* demangle( const char* module,
                      const char* function )
  {
  Bsc *bsc ;
  if( Bsc :: open( module, &bsc ) )
    return( bsc->formatDname( function ) ) ;
  else
    return( function ) ;
  }
