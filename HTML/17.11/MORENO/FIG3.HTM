struct Low_pass
{
    double Fc;      // Cutoff frequency
    double Fs;      // Sampling frequency
    int N;          // number of non-repeated coefficients

                // Constructor
    Low_pass (double fc, double fs, int n)
        : Fc(fc), Fs(fs), N(n) {}
};
