#include <stdio.h>
void main()
   {
    FILE *file_printer;
    int x, y, z;

    x = 5;
    y = 6;
    z = x + y;
    if ( (file_printer = fopen("LPT1", "w") ) == NULL)
        {
        printf ("unable to open file".\n");
        exit(0);
        }

   fprintf(file_printer, "The answer is %10d.", z);
   fflush (file_printer);
   fclose (file_printer);
   }
