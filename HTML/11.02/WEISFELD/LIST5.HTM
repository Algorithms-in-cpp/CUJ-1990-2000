void build_basis(void)
{
       int i,j;

       i = 1;

       /* create the labels for the orig tableau */
       for (j=1; j<VARIABLES+1; j++) {
              strcpy (var, "X");
              itoa(i, num);
              strcat (var,num);
              strcpy (objective[j], var);
              i++;
       }
       i = 1;
       for (j=VARIABLES+1; j<EQUATIONS+VARIABLES+1;
                           j++) {

              itoa(i, num);

              i++;
       }
       strcpy (objective[j], "sol");
       strcpy (var, "z");

}

void print_table(void)
{
       int i,j;
       printf ("\n");
       printf ("%6s"," basis |");
       for (j=0; j<=COLUMNS; j++) {
              if ((j==1) || (j==COLUMNS-1))
                    printf (" |");
              printf (" %6s", objective[j]);
       }
       printf ("\n");
       for (j=0; j<COLUMNS+2; j++) {
              printf ("-------");
       }
       printf ("\n");
       for (i=0; i<ROWS;i++) {
              printf (" %6s", basis[i]);
              for (j=0; j<COLUMNS; j++) {
                     if ((j==0) || (j==1) ||
                           (j==COLUMNS-1))
                           printf (" |");
                     printf (" %6.2f",table[i][j]);
              }
              if (i==0) {
                     printf ("\n");
                     for (j=0; j<COLUMNS+2; j++) {
                           printf ("-------");
                     }
              }
              printf ("\n");
       }
       printf ("\n");
}
