// tdate7.cpp:

#include <iostream.h>
#include "date7.h"

main()
{
   Date d1, d2;
   cout << "Enter a date: ";
   cin >> d1;
   cout << "Enter another date: ";
   cin >> d2;
   cout << "d1 - d2 = "<< d1 - d2 << endl;
   cout << "d2 - d1 = "<< d2 - d1 << endl;
   return 0;
}

/* OUTPUT

Enter a date: 10/1/1951
Enter another date: 5/1/1954
d1 - d2 = -7/0/-2
d2 - d1 = 7/0/2
