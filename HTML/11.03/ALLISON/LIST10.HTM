Listing 10 Implements Date stream I/O functions

#include <iostream.h>
#include "date7.h"

ostream& operator<<(ostream& os, const Date& d)
{
   os << d.month << '/' << d.day << '/' << d.year;
   return os;
}

istream& operator>>(istream& is, Date& d)
{
   char slash;
   is >> d.month >> slash >> d.day >> slash >> d.year;
   return is;
}
