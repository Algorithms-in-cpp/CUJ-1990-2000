Listing 1 Overriding an inherited function

#include <iostream.h>

class B
       {
public:
       void f();
       void g();
       };

void B::f() { cout << "B::f()\n"; }

void B::g() { cout << "B::g()\n"; }

class D : public B
       {
public:
       void g();
       };

void D::g() { cout << "D::g()\n"; }

int main()
       {
       B b;
       b.f();
       b.g();
       D d;
       d.f();
       d.g();
       return 0;
       }
