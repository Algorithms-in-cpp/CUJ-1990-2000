Listing 4 Illustrates the semantics of an exception specification

// The following is equivalent to:
// void f() throw(A,B) {//Whatever}

void f()
{
   try
   {
      // Whatever
   }
   catch(A)
   {
      throw;        // rethrow
   }
   catch(B)
   {
      throw;        // rethrow
   }
   catch(...)
   {
      unexpected( );
   }
}
