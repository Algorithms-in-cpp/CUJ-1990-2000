int main()
    {
    char word[MAXWORD];
    unsigned lineno = 1;
    xrt x;

    while (getword(word, MAXWORD) != EOF)
        if (isalpha(word[0]))
            x.add(word, lineno);
        else if (word[0] == '\n')
            ++lineno;
    x.print();
    return 0;
    }
