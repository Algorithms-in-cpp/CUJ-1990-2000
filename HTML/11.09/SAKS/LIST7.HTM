#include <iostream.h>

class File
       {
public:
       File() { ok = FALSE; }
       bool good() { return ok; }
       void reset(bool b = TRUE) { ok = bool(b); }
private:
       enum bool {FALSE, TRUE};
       bool ok;
       };

int main()
       {
       File f;
       cout << f.good() << '\n';
       cout << sizeof(f.good()) << '\n';
       cout << sizeof(f) << '\n';
       return 0;
       }
