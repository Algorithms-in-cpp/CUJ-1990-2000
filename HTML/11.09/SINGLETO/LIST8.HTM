winstreambuf::winstreambuf()
   : streambuf (buffer, winstreambuf_buf_size)
{

*buffer = 0;
reset_buffer ();
stream_window = NULL;
}

int winstreambuf::sync ()
{
ASSERT (stream_window != NULL);
stream_window->PutText (pbase (), out_waiting ());
reset_buffer ();
return 0;
}
