void Account::entry_post(Entry * pentry)
   {
   switch(type_of(pentry))
      {
   case Deposit_entry:
      balance += pentry->amount;
      break;
   case Check_entry:
      balance -= pentry->amount;
      break;
   case Withdrawal_entry:
      balance -= pentry->amount;
      break;
      }
