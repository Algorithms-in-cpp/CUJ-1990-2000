_TEXT segment word public 'CODE'

; WHICH_MOUSE - Get driver revision, mouse type,
; and IRQ number
;
; entry:       nothing
;
; exit:             BH = major revision number in BCD
;                   BL = minor revision number in BCD
;                   CH = mouse type 1-5
;                   CL = IRQ number 0, or 2-5, or 7
;
; assembler:        MASM 4.0+

Which_Mouse proc near
    public Which_Mouse
    
    mov ax, 36
    int 33h
    ret

Which_Mouse endp

_TEXT     ends
