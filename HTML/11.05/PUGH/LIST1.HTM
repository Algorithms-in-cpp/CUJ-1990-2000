#include <fcntl.h>
#include <sys\stat.h>
#include <dos.h>
#include <io.h>
#include <stdio.h>
void main()
       {
       int file;
       int count_written;
       file =_open ("TEMP", O_RDWR);
       if (file < 0)
              {
              printf("Error %d in file open\n", errno);
              perror("");
              )
       count_written = _write(file,&file,0);
       printf("Count written %d", count_written);
       _close(file);
       }
