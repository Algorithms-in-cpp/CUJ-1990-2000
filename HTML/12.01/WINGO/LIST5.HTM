AFX_INLINE CArchive& CArchive::operator<<(DWORD dw)
{
       if (m_lpBufCur + sizeof(DWORD) > m_lpBufMax)
                Flush();

        memcpy(m_lpBufCur,&dw,sizeof(DWORD));
        
        m_lpBufCur += sizeof(DWORD);
        return *this;
}
