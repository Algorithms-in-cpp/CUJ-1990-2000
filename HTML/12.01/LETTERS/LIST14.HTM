// pizza.cpp
#include <iostream.h>
#include <string.h>

struct PIZZA
{
   int key;
   char stuff[10];
};

main()
{
   PIZZA  *myPizza = new PIZZA[10];
   myPizza->key = 0;
   strcpy(myPizza->stuff,"good food");
   cout << myPizza[0].key << ": "<< myPizza[0].stuff << endl;
   return 0;
}
