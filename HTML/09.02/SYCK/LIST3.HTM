Listing 3 Non-Recursive Tree Search Routine

int
TreeSearch( int NodeNumb, int Value )
{
    while( NodeNumb != -1 && Value
                 ! = Tree[NodeNumb].Data )
    {
          if( Value < Tree[NodeNumb].Data )
                 NodeNumb = Tree[NodeNumb].Left;
          else
                 NodeNumb = Tree[NodeNumb].Right;
    }
}
