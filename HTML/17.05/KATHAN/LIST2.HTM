Listing 2: A Unix daemon killer

#include <stdlib.h>
#include <stdio.h>
#include <signal.h>
     
#define LOGFILE "pid.log"
     
main(int argc, char* argv[])
{
FILE *fp;
pid_t pid;
     
fp = fopen(LOGFILE,"r");
while (!feof(fp))
    {
    fscanf(fp,"%u",&pid);
    if (!feof(fp))
        kill(pid,SIGTERM);
    }
fclose(fp);
     
exit(0);
}
