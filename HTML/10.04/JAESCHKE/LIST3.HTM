void push(stack *st, int value)
{
   if (st->stack_ptr == st->max_stack)
      printf("Stack %s is full\n", st->stack_name);
   else
      st->pstack[st->stack_ptr++] = value;
}

int pop(stack *st)
{
   if (st->stack_ptr == 0) {
      printf("Stack %s is empty\n", st->stack_name);
      return 0;
   }
   
   return st->pstack[-st->stack_ptr];
}
