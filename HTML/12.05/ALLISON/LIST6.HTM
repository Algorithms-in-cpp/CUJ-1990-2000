Listing 6 Defines a Local class

#include <iostream.h>

int i = 10;

main()
{
   void f();
   f();
   return 0;
}

void f()
{
   static int j = 20;
   
   class Local
   {
      int k;
   public:
      Local(int i) : k(i) {}
      void a() {cout << k+i << endl;}
      void b() {cout << k+j << endl;}
   };
   
   Local l[30];
   l.a();
   l.b();
}

40
50
