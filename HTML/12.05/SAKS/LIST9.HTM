#include "lns.h"

class lns_fast : public lns
   {
public:
   lns_fast(unsigned n);
   ~lns_fast();
   void add(unsigned n);
   void print();
private:
   struct node;
   node *first, *last;
   };

struct lns_fast::node
   {
   node(unsigned n);
   unsigned number;
   node *next;
   };
