Listing 2 Illustrates function and file scope

/* scope2.c */

#include <stdio.h>

main()
{
   void fl(int i);
   void f2(void);
   
   f1(23);
   f2();
   return 0;
}

int i = 13;

void fl(int i)
{
   for (;;)
   {
      float i = 33.0;
      
      printf("%f\n",i);
      goto exit;
   }

exit:
   printf("%d\n",i);
}

void f2(void)
{
   printf("%d\n",i);
}

/* Output:
33.000000
23
13
*/
