#include <stdio.h>

main()
{
       char *p1;
       char (*p2)[3];
       char (*p3)[6];

       printf("sizeof(*p1) = %lu\n",
              (unsigned long)sizeof(*p1));
       printf("sizeof(*p2) = %lu\n",
              (unsigned long)sizeof(*p2));
       printf("sizeof(*p3) = %lu\n",
              (unsigned long)sizeof(*p3));
}
