Listing 9 (reporter.c) Reporter Task

#include <synrtx.h>

task Reporter(void) {
   task_t self = task_self();
   msg_t msg;

   task_setHandler(0);
   loop {
      /* Wait for a while ... */
      task_delay(500L);

      /* ... then report */
      msg.srcTid = self;
      msg.dstTid = task_idOf("Update");
      msg.dstHid = 1; /* Report = 1 */
      msg.type    = msg_type_SYNC;
      task_interrupt(&msg);
   }
}
