unsigned _stdcall srv_cPipeServer::ServerThread(void *pThreadData_)
{
    printf("Starting thread %u\n", GetCurrentThreadId());
    srv_cThread *pThread = (srv_cThread *) pThreadData_;
    pThread -> Handler();
    delete pThread;
    printf("Ending thread %u\n", GetCurrentThreadId());
    return 0;
}
