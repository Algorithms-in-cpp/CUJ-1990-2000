Listing 2 pad.c — pads an existing string with blanks via a variable-length format

/* pad.c:   Add trailing blanks to a string */

#include <stdio.h>
#include <string.h>

char *pad(char *s, int size)
{
   int len = strlen(s);
   if (s != NULL && len < size)
      sprintf(s+len,"%*c",size-len,' ');
   return s;
}
