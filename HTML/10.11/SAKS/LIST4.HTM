#include <iostream.h>
#include "fa1.h"

void display(char *s, float_array &fa)
   {
   cout << s << " = "<< fa << '\n';
   }

float sigma(float_array &fa)
   {
   size_t i;
   float sum = 0;
   for (i = 0; i < fa.length(); ++i)
      sum += fa[i];
   return sum;
   }

int main()
   {
   size_t i, size;
   cout << "size? ";
   cin >> size;

   float_array fa(size), fb;
   display("fa", fa);
   display("fb", fb);
   for (i = 0; i < fa.length(); ++i)
      fa[i] = i;
   display("fa", fa);
   float_array fc = fa;
   display("fc", fc);
   for (i = 0; i < fc.length(); ++i)
      fc[i] = i * i;
   display("fc", fc);
   cout << "sigma(fc) = " << sigma(fc) << '\n';
   return 0;
   }
