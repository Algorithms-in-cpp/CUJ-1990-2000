#ifndef JMPSTACK_H
#define JMPSTACK_H

#include <setjmp.h>

class JmpStack {
    enum {SIZE=100};
    jmp_buf stack[SIZE];
    int current;

public:
    JmpStack() {current = -1;}
    
    jmp_buf& operator++(); // push
    jmp_buf& operator--(); // pop
};

#endif
