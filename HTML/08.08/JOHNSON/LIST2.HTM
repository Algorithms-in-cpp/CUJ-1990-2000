   enum    adaptor
          {
          UnknownAdaptor,
          MDA,
          CGA,
          EGA,
          MCGA,
          VGA,
          HGC,
          HGCPlus,
          HerculesInColor
          };
   
   enum    monitor
          {
          UnknownMonitor,
          MDAMonochrome,
          CGAColor,
          EGAColor,
          PS2Monochrome,
          PS2Color
          };
   
   struct video
          {
          enum adaptor   VideoAdaptor;
          enum monitor   VideoMonitor;
          };
 
   extern struct video    *IdentifyVideo(void);
